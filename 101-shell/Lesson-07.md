`echo OneOneOne > one`

`echo TwoTwoTwo > two`

`cat one`

`cat two`

`cat one two`

`cat one two > onetwo`

`cat onetwo`

*Note cat combined one and two into onetwo*

`echo ThreeThreeThree > three`

`cat three two one >> onetwo`

`cat onetwo`

*Note cat also appended to onetwo via >>*

`rm one two three onetwo`
